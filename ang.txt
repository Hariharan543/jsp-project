export class AppComponent {
title='Age!';
count=false;
age: number;
getage(value)
{
  var today = new Date();
  var birthDate = new Date(value);
  var age = today.getFullYear() - birthDate.getFullYear();
  var m = today.getMonth() - birthDate.getMonth();
  if (m < 0 || (m === 0 && today.getDate() < birthDate.getDate())) {
      age--;
  }

  this.age=age;
  this.count=true;    
   }
}
